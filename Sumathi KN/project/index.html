Pre-Processing
Prompt
Map all NSL-KDD attack sub-labels into 4 main classes (DoS, Probe, R2L, U2R) and keep Normal. Apply One-Hot Encoding to protocol_type, service and flag columns. Align train and test columns. Apply Min-Max normalisation. Print class distribution after mapping.
Python
#Map attack sub-labels to 4 main attack classes
dos_attacks   = ['neptune','back','land','pod','smurf','teardrop',
                 'mailbomb','apache2','processtable','udpstorm']
probe_attacks = ['ipsweep','nmap','portsweep','satan','mscan','saint']
r2l_attacks   = ['ftp_write','guess_passwd','imap','multihop','phf','spy',
                 'warezclient','warezmaster','sendmail','named',
                 'snmpgetattack','snmpguess','xlock','xsnoop','httptunnel']
u2r_attacks   = ['buffer_overflow','loadmodule','perl','rootkit',
                 'ps','sqlattack','xterm']

def map_label(label):
    label = label.strip().lower()
    if label in dos_attacks:    return 'DoS'
    if label in probe_attacks:  return 'Probe'
    if label in r2l_attacks:    return 'R2L'
    if label in u2r_attacks:    return 'U2R'
    return 'Normal'

for df in [train_df, test_df]:
    df['attack_class'] = df['label'].apply(map_label)

print("Class distribution after mapping (Train):")
print(train_df['attack_class'].value_counts().to_string())
Class distribution after mapping (Train): Normal 67343 DoS 45927 Probe 11656 R2L 995 U2R 52 ← severe imbalance
Python
#One-Hot Encoding for categorical columns
cat_cols = ['protocol_type', 'service', 'flag']
train_df = pd.get_dummies(train_df, columns=cat_cols)
test_df  = pd.get_dummies(test_df,  columns=cat_cols)

#Align columns — fill missing test columns with 0
train_df, test_df = train_df.align(test_df, join='left', axis=1, fill_value=0)

drop_cols   = ['label', 'attack_class']

y_test      = test_df['attack_class']

#Min-Max Normalisation
scaler     = MinMaxScaler()
X_train_sc = scaler.fit_transform(X_train_raw)
X_test_sc  = scaler.transform(X_test_raw)

print(f"Features after One-Hot Encoding : {X_train_raw.shape[1]}")

    model_classes  = list(model.classes_)
    y_prob_aligned = np.zeros((len(y_test), len(classes_list)))
    for i, cls in enumerate(classes_list):
        if cls in model_classes:
            j = model_classes.index(cls)
            y_prob_aligned[:, i] = y_prob[:, j]
    auc_list = []
    for cidx, cls in enumerate(classes_list):
        fpr, tpr, _ = roc_curve(y_test_bin_all[:, cidx], y_prob_aligned[:, cidx])
        auc_v = auc(fpr, tpr); auc_list.append(auc_v)
        ax.plot(fpr, tpr, linestyle=c_styles[cidx%len(c_styles)],
                color=c_colors[cidx], linewidth=2.2, label=f'{cls} (AUC={auc_v:.3f})')
    ax.plot([0,1],[0,1],'k--',linewidth=1,alpha=0.5,label='Random (0.5)')
    ax.set_title(f'{name}\\nMean AUC={np.mean(auc_list):.4f}', fontsize=12, fontweight='bold', color=tc)
    ax.set_xlabel('False Positive Rate'); ax.set_ylabel('True Positive Rate')
    ax.legend(loc='lower right', fontsize=8.5)
