Step 13: K-Fold Cross-Validation for KNN and Random Forest

Code

from sklearn.model_selection import StratifiedKFold

# StratifiedKFold ensures each fold has approximately the same class distribution
kf = StratifiedKFold(n_splits=5, shuffle=True, random_state=42)

knn_accuracies = []
rf_accuracies = []

# Iterate over each fold
for train_index, test_index in kf.split(X, y):
    X_train_fold, X_test_fold = X.iloc[train_index], X.iloc[test_index]
    y_train_fold, y_test_fold = y[train_index], y[test_index]

    # Train and evaluate KNN for the current fold
    knn_cv = KNeighborsClassifier(n_neighbors=3)
    knn_cv.fit(X_train_fold, y_train_fold)
    y_pred_knn_cv = knn_cv.predict(X_test_fold)
    knn_accuracies.append(accuracy_score(y_test_fold, y_pred_knn_cv))

    # Train and evaluate Random Forest for the current fold
    rf_cv = RandomForestClassifier(n_estimators=100, random_state=42)
    rf_cv.fit(X_train_fold, y_train_fold)
    y_pred_rf_cv = rf_cv.predict(X_test_fold)
    rf_accuracies.append(accuracy_score(y_test_fold, y_pred_rf_cv))

print(f"Average KNN Accuracy (5-fold CV): {np.mean(knn_accuracies):.4f}")
print(f"Standard Deviation KNN Accuracy (5-fold CV): {np.std(knn_accuracies):.4f}")
print(f"Average Random Forest Accuracy (5-fold CV): {np.mean(rf_accuracies):.4f}")
print(f"Standard Deviation Random Forest Accuracy (5-fold CV): {np.std(rf_accuracies):.4f}")

Output

Average KNN Accuracy (5-fold CV): 0.9890
Standard Deviation KNN Accuracy (5-fold CV): 0.0058
Average Random Forest Accuracy (5-fold CV): 0.9910
Standard Deviation Random Forest Accuracy (5-fold CV): 0.0037

K-Fold Cross-Validation for KNN with Scaled Features

Code

# Apply scaling within each fold to prevent data leakage
knn_scaled_accuracies_cv = []

for train_index, test_index in kf.split(X, y):
    X_train_fold, X_test_fold = X.iloc[train_index], X.iloc[test_index]
    y_train_fold, y_test_fold = y[train_index], y[test_index]

    # Scale features for the current fold
    scaler_cv = StandardScaler()
    X_train_fold_scaled = scaler_cv.fit_transform(X_train_fold)
    X_test_fold_scaled = scaler_cv.transform(X_test_fold)

    # Train and evaluate KNN with scaled data
    knn_cv_scaled = KNeighborsClassifier(n_neighbors=3)
    knn_cv_scaled.fit(X_train_fold_scaled, y_train_fold)
    y_pred_knn_cv_scaled = knn_cv_scaled.predict(X_test_fold_scaled)
    knn_scaled_accuracies_cv.append(accuracy_score(y_test_fold, y_pred_knn_cv_scaled))

print(f"Average KNN Accuracy (5-fold CV, Scaled): {np.mean(knn_scaled_accuracies_cv):.4f}")
print(f"Standard Deviation KNN Accuracy (5-fold CV, Scaled): {np.std(knn_scaled_accuracies_cv):.4f}")
