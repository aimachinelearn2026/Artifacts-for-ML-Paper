# 5. PREPROCESSING
# =====================================================
stop_words = set(stopwords.words('english'))

def preprocess(text):
    text = re.sub('[^a-zA-Z]', ' ', str(text))   # remove special chars
    text = text.lower()                              # lowercase
    words = text.split()                              # tokenize
    words = [w for w in words if w not in stop_words] # remove stopwords
    return " ".join(words)

# Combine title + text into one 'content' column
df['content'] = df['title'] + " " + df['text']
df['content'] = df['content'].apply(preprocess)
Preprocessing pipeline: regex cleanup → lowercase → stopword removal. Title and body text are merged before cleaning to give the model more signal.

PYTHON · CELL 6 
# 6. TF-IDF
# =====================================================
vectorizer = TfidfVectorizer(max_features=5000)
X = vectorizer.fit_transform(df['content'])
y = df['label']
TF-IDF (Term Frequency–Inverse Document Frequency) gives each word a score based on how often it appears in a document vs. the entire corpus. max_features=5000 caps the vocabulary size to the top 5000 words.

models = {
    "Logistic Regression": LogisticRegression(),
    "Naive Bayes": MultinomialNB(),
    "Decision Tree": DecisionTreeClassifier(),
    "SVM": SVC(probability=True)

accuracy_list = []
model_names = []

for name, model in models.items():
    model.fit(X_train, y_train)
    y_pred = model.predict(X_test)

    acc = accuracy_score(y_test, y_pred)
    accuracy_list.append(acc)
    model_names.append(name)

    print(f"\n{name} Accuracy:", acc)
    print(classification_report(y_test, y_pred))

🤖 AI Prompt Used

"Write a predict_news(text) function that takes raw input text, preprocesses it, vectorizes using the trained TF-IDF vectorizer, predicts using the Logistic Regression model, and returns 'Fake News ❌' or 'Real News ✅'."

PYTHON · CELL 9 
# 9. PREDICTION FUNCTION
# =====================================================
def predict_news(text):
    text = preprocess(text)
    vec = vectorizer.transform([text])
    pred = lr.predict(vec)
    return "Fake News ❌" if pred[0] == 1 else "Real News ✅"

print("\nPrediction Example:")
print(predict_news("Government launches new education scheme"))
print(predict_news("Aliens have landed in New Delhi today"))
