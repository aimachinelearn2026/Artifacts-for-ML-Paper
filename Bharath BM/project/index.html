class AlexNetConfig(nn.Module):
    """
    AlexNet adapted for MNIST (64x64x3 input, 10 classes)
    Supports 4 configurations via flags:
      - use_relu      : ReLU vs Tanh activation
      - use_dropout   : Dropout(0.5) in FC layers
    """
    def __init__(self, use_relu=True, use_dropout=True, num_classes=10):
        super(AlexNetConfig, self).__init__()
        act = nn.ReLU(inplace=True) if use_relu else nn.Tanh()

        self.features = nn.Sequential(
            # Conv Block 1
            nn.Conv2d(3, 64, kernel_size=5, stride=2, padding=2),
            act,
            nn.MaxPool2d(kernel_size=3, stride=2),
            # Conv Block 2
            nn.Conv2d(64, 128, kernel_size=5, padding=2),
            act,
            nn.MaxPool2d(kernel_size=3, stride=2),
            # Conv Block 3
            nn.Conv2d(128, 256, kernel_size=3, padding=1),
            act,
            # Conv Block 4
            nn.Conv2d(256, 256, kernel_size=3, padding=1),
            act,
            # Conv Block 5
            nn.Conv2d(256, 128, kernel_size=3, padding=1),
            act,
            nn.MaxPool2d(kernel_size=3, stride=2),
        )

        self.classifier = nn.Sequential(
            nn.Dropout(0.5) if use_dropout else nn.Identity(),
            nn.Linear(128 * 3 * 3, 1024),
            act,
            nn.Dropout(0.5) if use_dropout else nn.Identity(),
            nn.Linear(1024, 512),
            act,
            nn.Linear(512, num_classes),
        )

    def forward(self, x):
        x = self.features(x)
        x = torch.flatten(x, 1)
        x = self.classifier(x)
        return x

def train_model(config_name, model_kwargs, use_augmentation=False, epochs=10, lr=0.001):
    print(f'\n{"="*60}')
    print(f'  Training: {config_name}')
    print(f'{"="*60}')

    model = AlexNetConfig(**model_kwargs).to(DEVICE)
    optimizer = optim.Adam(model.parameters(), lr=lr)
    scheduler = optim.lr_scheduler.StepLR(optimizer, step_size=5, gamma=0.5)
    criterion = nn.CrossEntropyLoss()

    loader = train_loader_aug if use_augmentation else train_loader_base

    history = {'train_loss': [], 'train_acc': [], 'test_acc': []}

    for epoch in range(1, epochs + 1):
        # --- Train ---
        model.train()
        running_loss, correct, total = 0.0, 0, 0
        for imgs, labels in loader:
            imgs, labels = imgs.to(DEVICE), labels.to(DEVICE)
            optimizer.zero_grad()
            outputs = model(imgs)
            loss = criterion(outputs, labels)
            loss.backward()
            optimizer.step()
            running_loss += loss.item() * imgs.size(0)
            _, preds = outputs.max(1)
            correct += preds.eq(labels).sum().item()
            total += labels.size(0)
        train_loss = running_loss / total
        train_acc  = 100.0 * correct / total

        # --- Evaluate ---
        model.eval()
        t_correct, t_total = 0, 0
        with torch.no_grad():
            for imgs, labels in test_loader:
                imgs, labels = imgs.to(DEVICE), labels.to(DEVICE)
                outputs = model(imgs)
                _, preds = outputs.max(1)
                t_correct += preds.eq(labels).sum().item()
                t_total += labels.size(0)
        test_acc = 100.0 * t_correct / t_total

        history['train_loss'].append(train_loss)
        history['train_acc'].append(train_acc)
        history['test_acc'].append(test_acc)

        scheduler.step()
        print(f'  Epoch {epoch:02d}/{epochs} | Loss: {train_loss:.4f} | '
              f'Train Acc: {train_acc:.2f}% | Test Acc: {test_acc:.2f}%')

    print(f'  ✅ Final Test Accuracy: {history["test_acc"][-1]:.2f}%')
    return model, history
