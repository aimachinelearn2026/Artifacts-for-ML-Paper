Loading the dataset, cleaning constant columns, and preparing the custom multi-class mapping.

import pandas as pd
import numpy as np
from sklearn.model_selection import train_test_split
from sklearn.preprocessing import MinMaxScaler, LabelEncoder

# 1.1 Load dataset
file_path = '/content/drive/MyDrive/Colab Notebooks/MalMem2022.csv'
df = pd.read_csv(file_path)

# 1.2 Basic cleaning: remove constant columns
df = df.loc[:, df.nunique() > 1]

# 1.3 Custom mapping for multi-label classification
def map_category_to_multilabel(category_name):
    if 'Benign' in category_name: return 0
    elif 'Ransomware' in category_name: return 1
    elif 'Spyware' in category_name: return 2
    elif 'Trojan' in category_name: return 3
    else: return -1

y_binary = LabelEncoder().fit_transform(df['Class'])
y_multi = df['Category'].apply(map_category_to_multilabel).values

# 1.4 Feature Selection (Numeric only)
X = df.select_dtypes(include=[np.number]).drop(columns=['Class', 'Category'], errors='ignore')

# 1.5 Train-Test Split (80/20)
X_train, X_test, y_train_bin, y_test_bin, y_train_multi, y_test_multi = train_test_split(
    X, y_binary, y_multi, test_size=0.2, stratify=y_multi, random_state=42
)

# 1.6 Global Scaling
scaler = MinMaxScaler()
X_train_scaled = scaler.fit_transform(X_train)
X_test_scaled = scaler.transform(X_test)

print(f"Training set shape: {X_train_scaled.shape}")

Training set shape: (46876, 52)

3. Robust Evaluation: 5-Fold Cross-Validation¶

Validating the model using a Pipeline to ensure no data leakage.

In [3]:
from sklearn.model_selection import StratifiedKFold, cross_validate
from sklearn.pipeline import Pipeline
from sklearn.ensemble import RandomForestClassifier

# 3.1 Setup Stratified 5-Fold
skf = StratifiedKFold(n_splits=5, shuffle=True, random_state=42)

# 3.2 Define Pipelines (Scale + Model)
bin_pipe = Pipeline([('scaler', MinMaxScaler()), ('rf', RandomForestClassifier(n_estimators=200, max_depth=20, n_jobs=-1, random_state=42))])
mul_pipe = Pipeline([('scaler', MinMaxScaler()), ('rf', RandomForestClassifier(n_estimators=300, max_depth=25, n_jobs=-1, random_state=42))])

print("Running 5-Fold Cross-Validation...")
bin_res = cross_validate(bin_pipe, X_train_cv, y_train_bin, cv=skf, scoring=['accuracy', 'f1_macro'])
mul_res = cross_validate(mul_pipe, X_train_cv, y_train_multi, cv=skf, scoring=['accuracy', 'f1_macro'])

print(f"Binary - Mean Acc: {bin_res['test_accuracy'].mean()*100:.2f}% | Mean F1: {bin_res['test_f1_macro'].mean()*100:.2f}%")
print(f"Multi  - Mean Acc: {mul_res['test_accuracy'].mean()*100:.2f}% | Mean F1: {mul_res['test_f1_macro'].mean()*100:.2f}%")
