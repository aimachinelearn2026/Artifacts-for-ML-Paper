def find_parquet_files():
    """Search common locations for parquet files."""
    search_patterns = [
        '*.parquet',
        '/content/*.parquet',
        '/content/drive/MyDrive/*.parquet',
    ]

    for pattern in search_patterns:
        found.extend(glob.glob(pattern))

def preprocess(df):
    """Encode categoricals, drop non-feature columns."""
    df = df.copy()
    # Drop non-feature columns
    drop_these = [c for c in ['id', 'attack_cat'] if c in df.columns]
    df.drop(columns=drop_these, inplace=True)
    # Label encode string columns
    le = LabelEncoder()
    for col in df.select_dtypes(include=['object', 'category']).columns:
        if col != 'label':
            df[col] = le.fit_transform(df[col].astype(str))
    return df
